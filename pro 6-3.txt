import java.util.Scanner;
class Divisible_by_7_13
{
    public static void main(String[] args)
    {
        Scanner input = new Scanner(System.in);
        System.out.print("Enter the Number : ");
        int num = input.nextInt();
        if(num % 7 == 0 && num % 13 == 0)
        {
            System.out.println("The number is divisible by both 7 and 13.");
            System.out.println("The quotient when the number is divided by 7 is : "+(num / 7));
            System.out.println("The remainder when the number is divided by 7 is : "+(num % 7));
            System.out.println("The quotient when the number is divided by 13 is : "+(num / 13));
            System.out.println("The remainder when the number is divided by 13 is : "+(num % 13));
        }
        else
        {
            System.out.println("The number is not divisible by both 7 and 13.");
        }
    }
}
